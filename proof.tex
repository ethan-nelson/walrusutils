\begin{proof}
  Let $a,b \in \mathbb{R} ^2$ where $a = (x_1,y_1)$, $b = (x_2,y_2)$, and $x_1 \neq x_2$. So,
  \begin{align*}
    (y-y_1) &= m (x-x_1), &&    m = \frac{y_2-y_1}{x_2-x_1} \\
    y &= \frac{y_2-y_1}{x_2-x_1} (x - x_1) + y_1&& \text{(combining equations)} \\
    &= \frac{y_2-y_1}{x_2-x_1}x - \frac{x_1 y_2 - x_1 y_1}{x_2 - x_1} + y_1 \\
    &= \frac{y_2-y_1}{x_2-x_1}x - \frac{x_1 y_2 - x_1 y_1}{x_2 - x_1} + \frac{x_2 - x_1}{x_2 - x_1}y_1 \\
    &= \frac{y_2-y_1}{x_2-x_1}x - \frac{x_1 y_2 - x_1 y_1}{x_2 - x_1} + \frac{x_2 y_1 - x_1 y_1}{x_2 - x_1} \\
    &= \frac{y_2-y_1}{x_2-x_1}x + \frac{- x_1 y_2 + x_1 y_1 + x_2 y_1 - x_1 y_1}{x_2 - x_1} \\
    &= \frac{(y_2-y_1)x + x_2 y_1 - x_1 y_2}{x_2 - x_1} && \qedhere
  \end{align*}
\end{proof}
